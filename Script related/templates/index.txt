'use strict';

// ── Utilities ────────────────────────────────────────────────────────────
function esc(s) {
  return String(s)
    .replace(/&/g,'&amp;').replace(/</g,'&lt;')
    .replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}
function closeOv(id, e) {
  if (!e || e.target.id === id)
    document.getElementById(id).classList.remove('show');
}
function toast(msg, isErr = false) {
  const el = document.getElementById('toast');
  el.textContent = msg;
  el.className = 'toast' + (isErr ? ' err' : '') + ' show';
  setTimeout(() => el.classList.remove('show'), 2800);
}
function switchPanel(name) {
  ['t2f','trex'].forEach(p =>
    document.getElementById(p + 'Panel').classList.toggle('active', p === name));
  document.querySelectorAll('.main-tab').forEach((b, i) =>
    b.classList.toggle('active', ['t2f','trex'][i] === name));
}
document.addEventListener('keydown', e => {
  if (e.key === 'Escape')
    document.querySelectorAll('.overlay.show').forEach(o => o.classList.remove('show'));
});

// ── Status panel ─────────────────────────────────────────────────────────
let _lastStatus = null;

function applyStatusToPill(data) {
  const pill = document.getElementById('folderPill');
  const txt  = document.getElementById('folderPillText');
  if (!data.folder_ok) {
    pill.className = 'folder-pill err';
    txt.textContent = `❌ "${data.folder}" not found`;
  } else if (data.err_count > 0) {
    pill.className = 'folder-pill err';
    txt.textContent = `📁 ${data.ok_count} ok, ${data.err_count} failed`;
  } else {
    pill.className = 'folder-pill ok';
    txt.textContent = `📁 ${data.ok_count} files loaded`;
  }
}

function openStatus() {
  if (!_lastStatus) return;

function renderStatusModal(data) {
  const TREX_KEYS = ['Keywords CSV','Definition - Inclusion CSV',
    'Definition - Exclusion CSV','Cheat codes CSV','Ambiguous CSV'];
  const FEES_KEY  = 'Fees Percentage csv';

  document.getElementById('statusTitle').textContent =
    `📁 "${data.folder}" — File Status`;
  document.getElementById('statusSubtitle').textContent =
    `${data.ok_count} loaded successfully · ${data.err_count} failed`;

  const body = document.getElementById('statusBody');
  body.innerHTML = '';

  // TRexID section
  const trexFiles = Object.entries(data.files)
    .filter(([k]) => TREX_KEYS.includes(k) || k === FEES_KEY);
  if (trexFiles.length) {
    const sec = document.createElement('div');
    sec.innerHTML = `<div class="status-section-title">TRexID & Fees Files</div>
      <div class="status-grid" id="sg-trex"></div>`;
    body.appendChild(sec);
    const grid = sec.querySelector('#sg-trex');
    trexFiles.forEach(([name, status]) => {
      grid.appendChild(makeStatusItem(name, status));
    });
  }

  // T2F section
  const t2fFiles = Object.entries(data.files)
    .filter(([k]) => !TREX_KEYS.includes(k) && k !== FEES_KEY && k !== '_folder');
  if (t2fFiles.length) {
    const sec = document.createElement('div');
    sec.innerHTML = `<div class="status-section-title">T2F Marketplace Files (${t2fFiles.length})</div>
      <div class="status-grid" id="sg-t2f"></div>`;
    body.appendChild(sec);
    const grid = sec.querySelector('#sg-t2f');
    t2fFiles.forEach(([name, status]) => {
      grid.appendChild(makeStatusItem(name, status));
    });
  }
}

function makeStatusItem(name, status) {
  const isOk  = status.startsWith('✓');
  const count = isOk ? status : status.replace('❌ ', '');
  const el = document.createElement('div');
  el.className = `status-item ${isOk ? 'ok' : 'err'}`;
  el.innerHTML = `
    <span>${isOk ? '✓' : '✗'}</span>
    <span class="si-name" title="${esc(name)}">${esc(name)}</span>
    <span class="si-count">${esc(count)}</span>`;
  return el;
}

async function reloadFiles() {
  const btn = document.getElementById('reloadBtn');
  btn.disabled = true;
  btn.innerHTML = '<span class="spinning">⟳</span> Reloading…';

  const res = await fetch('/api/reload', { method: 'POST' })
    .then(r => r.json()).catch(() => null);

  btn.disabled = false;
  btn.textContent = '🔄 Reload';

  if (!res) { toast('Reload failed', true); return; }

  _lastStatus = { ...res, folder: _lastStatus?.folder || '', folder_ok: true };
  applyStatusToPill(_lastStatus);

  // Refresh marketplace dropdown
  const sel  = document.getElementById('mpSelect');
  const prev = sel.value;
  while (sel.options.length > 1) sel.remove(1);
  res.marketplaces.forEach(mp => {
    const o = document.createElement('option');
    o.value = mp; o.text = mp;
    sel.appendChild(o);
  });

  if (prev && res.marketplaces.includes(prev)) {
    sel.value = prev;
    await loadMpInfo(prev);
  } else if (res.marketplaces.length) {
    sel.value = res.marketplaces[0];
    await loadMpInfo(res.marketplaces[0]);
  }

  toast(`✓ Reloaded — ${res.ok_count} files loaded`);
}

function dT2F() {
  clearTimeout(t2fTimer);
  t2fTimer = setTimeout(runT2FSearch, 280);
}

async function runT2FSearch() {
  const { mp } = t2fState;
  if (!mp) return;

  const fields = Array.from(
    document.getElementById('t2fGrid').querySelectorAll('.sf'));
  const sv = [], fs_arr = [];
  fields.forEach(f => {
    const v = f.querySelector('input').value.trim();
    (f.dataset.isFees === '1' ? fs_arr : sv).push(v);
  });

  if (sv.every(v => !v) && !fs_arr[0]) {
    document.getElementById('t2fTableWrap').innerHTML = `<div class="empty">
      <div class="empty-ico">🔎</div>
      <div class="empty-h">Enter a search term</div>
      <div class="empty-p">Fill in one or more fields above to filter</div>
    </div>`;
    return;
  }

  const data = await fetch('/api/t2f/search', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ mp, search_values: sv, fees_search: fs_arr[0] || '' })
  }).then(r => r.json()).catch(() => null);

  if (!data) { toast('Search failed', true); return; }
  renderT2FTable(data.headers, data.rows);
}

function renderT2FTable(headers, rows) {
  const wrap = document.getElementById('t2fTableWrap');
  if (!rows.length) {
    wrap.innerHTML = `<div class="empty">
      <div class="empty-ico">🔎</div>
      <div class="empty-h">No results</div>
      <div class="empty-p">Try different search terms</div>
    </div>`;
    return;
  }

  let h = '<table><thead><tr>';
  headers.forEach(c => { h += `<th>${esc(c)}</th>`; });
  h += '</tr></thead><tbody>';
  rows.forEach(row => {
    h += '<tr>';
    headers.forEach(c => {
      const v = row[c] || '';
      if (c === 'Fees%')
        h += `<td class="${v === 'N/A' ? 'td-na' : 'td-fees'}">${esc(v)}</td>`;
      else
        h += `<td title="${esc(v)}">${esc(v)}</td>`;
    });
    h += '</tr>';
  });
  h += '</tbody></table>';
  wrap.innerHTML = h;
}

function openDef(idx, type, e) {
  if (e) e.preventDefault();
  const row = defStore[idx]; if (!row) return;
  const def  = type === 'incl' ? row.incl : row.excl;
  const term = document.getElementById('trexPkQ').value.trim().toLowerCase();

  let sents = def.split(/[.!?](?=\s|$)/).filter(s => s.trim());
  if (term) sents = sents.filter(s => s.toLowerCase().includes(term));

  const bullets = sents.map(s => {
    let t = esc(s.trim()) + '.';
    if (term) {
      const re = new RegExp(`(${term.replace(/[.*+?^${}()|[\]\\]/g,'\\$&')})`, 'gi');
      t = t.replace(re, '<mark>$1</mark>');
    }
    return `<li>${t}</li>`;
  }).join('');

  const badge = type === 'incl'
    ? `<span style="font-size:.7rem;color:var(--ok);font-weight:600;background:var(--ok-bg);border:1px solid var(--ok-bd);padding:2px 8px;border-radius:99px">Inclusion</span>`
    : `<span style="font-size:.7rem;color:var(--err);font-weight:600;background:var(--err-bg);border:1px solid var(--err-bd);padding:2px 8px;border-radius:99px">Exclusion</span>`;

  document.getElementById('defBody').innerHTML = `
    <h3>${esc(row.trexid)} ${badge}</h3>
    <ul>${bullets || '<li style="color:var(--t3)">No matching sentences found.</li>'}</ul>`;
  document.getElementById('defOverlay').classList.add('show');
}

function filterAmbi() {
  const qs = AMBI_KEYS.map((_, i) =>
    (document.getElementById(`ambiF_${i}`)?.value || '').toLowerCase());
  renderAmbi(ambiAll.filter(r =>
    AMBI_KEYS.every((k, i) => !qs[i] || r[k].toLowerCase().includes(qs[i]))));

  AMBI_KEYS.forEach((_, i) => {
    const el = document.getElementById(`ambiF_${i}`);
    if (el) el.value = '';
  });
  renderAmbi(ambiAll);

function renderAmbi(rows) {
  const tbody = document.getElementById('ambiTbody');
  tbody.innerHTML = '';
  const urlRe = /(https?:\/\/[^\s)]+)/g;

  rows.forEach(row => {
    const tr = document.createElement('tr');

    // ASIN
    const asinTd = document.createElement('td');
    asinTd.className = 'td-wrap'; asinTd.style.maxWidth = '200px'; asinTd.style.fontSize = '.74rem';
    const urls = row.asin.split(/[\n,]+/).map(u => u.trim()).filter(u => /^https?:\/\//i.test(u));
    if (urls.length) {
      urls.forEach((url, i) => {
        if (i > 0) asinTd.appendChild(document.createElement('br'));
        asinTd.appendChild(document.createTextNode(url + ' '));
        const a = document.createElement('a');
        a.href = url; a.textContent = 'Open'; a.target = '_blank'; a.style.color = 'var(--info)';
        asinTd.appendChild(a);
      });
    } else { asinTd.textContent = row.asin; }
    tr.appendChild(asinTd);

    // Description
    const dTd = document.createElement('td');
    dTd.className = 'td-wrap'; dTd.style.maxWidth = '280px'; dTd.style.fontSize = '.78rem';
    dTd.textContent = row.description; tr.appendChild(dTd);

    // TRexID
    const tTd = document.createElement('td');
    tTd.style.cssText = 'color:var(--info);font-weight:700;font-size:.78rem';
    tTd.textContent = row.trexid; tr.appendChild(tTd);

    // Comments (URL-linkified)
    const cTd = document.createElement('td');
    cTd.className = 'td-wrap'; cTd.style.maxWidth = '260px'; cTd.style.fontSize = '.74rem';
    let last = 0, match, text = row.comments;
    urlRe.lastIndex = 0;
    while ((match = urlRe.exec(text)) !== null) {
      cTd.appendChild(document.createTextNode(text.substring(last, match.index)));
      const a = document.createElement('a');
      a.href = match[1]; a.textContent = match[1]; a.target = '_blank'; a.style.color = 'var(--info)';
      cTd.appendChild(a);
      last = match.index + match[1].length;
    }
    cTd.appendChild(document.createTextNode(text.substring(last)));
    tr.appendChild(cTd);

    // Decision
    const decTd = document.createElement('td');
    decTd.className = 'td-wrap'; decTd.style.maxWidth = '140px'; decTd.style.fontSize = '.78rem';
    decTd.textContent = row.decision; tr.appendChild(decTd);

    tbody.appendChild(tr);
  });
}

function hlTerm(text, term) {
  if (!term) return text;
  const safe = term.replace(/[.*+?^${}()|[\]\\]/g,'\\$&');
  return text.replace(new RegExp(`(${safe})`, 'gi'), '<mark>$1</mark>');
}
