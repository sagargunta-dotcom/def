function buildDefTable(rows){

  let h=`<table><thead><tr><th>Incl (TRexID)</th><th>Excl (TRexID)</th></tr></thead><tbody>`;
  rows.forEach((r,i)=>{const iL=r.incl?`<a class="def-link-i" href="#" onclick="openDef(${i},'incl',event)">${esc(r.trexid)}</a>`:'';const eL=r.excl?`<a class="def-link-e" href="#" onclick="openDef(${i},'excl',event)">${esc(r.trexid)}</a>`:'';h+=`<tr><td>${iL}</td><td>${eL}</td></tr>`;});
  h+='</tbody></table>';w.innerHTML=h;return w;
}

function openDef(idx,type,e){
  if(e)e.preventDefault();const row=defStore[idx];if(!row)return;
  const def=type==='incl'?row.incl:row.excl,term=document.getElementById('trexPkQ').value.trim().toLowerCase();
  let sents=def.split(/[.!?](?=\s|$)/).filter(s=>s.trim());if(term)sents=sents.filter(s=>s.toLowerCase().includes(term));
  const bullets=sents.map(s=>{let t=esc(s.trim())+'.';if(term){const re=new RegExp(`(${term.replace(/[.*+?^${}()|[\]\\]/g,'\\$&')})`, 'gi');t=t.replace(re,'<mark>$1</mark>');}return`<li>${t}</li>`;}).join('');
  const badge=type==='incl'?`<span style="font-size:.68rem;color:var(--ok);font-weight:600;background:var(--ok-bg);border:1px solid var(--ok-bd);padding:2px 8px;border-radius:99px">Inclusion</span>`:`<span style="font-size:.68rem;color:var(--err);font-weight:600;background:var(--err-bg);border:1px solid var(--err-bd);padding:2px 8px;border-radius:99px">Exclusion</span>`;
  document.getElementById('defBody').innerHTML=`<h3>${esc(row.trexid)} ${badge}</h3><ul>${bullets||'<li style="color:var(--t3)">No matching sentences.</li>'}</ul>`;
  document.getElementById('defOverlay').classList.add('show');
}

/* ── Ambi ──────────────────────────────────────────────────────── */
const AMBI_KEYS=['asin','description','trexid','comments','decision'];
const AMBI_LBLS=['ASIN','Description','TRexID','Comments','Decision Pending?'];

async function showAmbi(){
  ambiAll=await fetch('/api/trex/ambi').then(r=>r.json()).catch(()=>[]);
  const row=document.getElementById('ambiSearchRow');row.innerHTML='';
  AMBI_LBLS.forEach((lbl,i)=>{const d=document.createElement('div');d.className='ambi-col';d.innerHTML=`<label>${lbl}</label><input type="text" id="ambiF_${i}" placeholder="Search…" oninput="filterAmbi()">`;row.appendChild(d);});
  const pk=document.getElementById('trexPkQ').value,tx=document.getElementById('trexIdQ').value;
  if(pk)document.getElementById('ambiF_1').value=pk;if(tx)document.getElementById('ambiF_2').value=tx;
  filterAmbi();document.getElementById('ambiOverlay').classList.add('show');
}
function filterAmbi(){const qs=AMBI_KEYS.map((_,i)=>(document.getElementById(`ambiF_${i}`)?.value||'').toLowerCase());renderAmbi(ambiAll.filter(r=>AMBI_KEYS.every((k,i)=>!qs[i]||r[k].toLowerCase().includes(qs[i]))));}
function clearAmbiFilters(){AMBI_KEYS.forEach((_,i)=>{const el=document.getElementById(`ambiF_${i}`);if(el)el.value='';});renderAmbi(ambiAll);}

function renderAmbi(rows){
  const tbody=document.getElementById('ambiTbody');tbody.innerHTML='';const urlRe=/(https?:\/\/[^\s)]+)/g;
  rows.forEach(row=>{
    const tr=document.createElement('tr');
    const asinTd=document.createElement('td');asinTd.className='td-wrap';asinTd.style.cssText='max-width:200px;font-size:.73rem';
    const urls=row.asin.split(/[\n,]+/).map(u=>u.trim()).filter(u=>/^https?:\/\//i.test(u));
    if(urls.length){urls.forEach((url,i)=>{if(i>0)asinTd.appendChild(document.createElement('br'));const a=document.createElement('a');a.href=url;a.textContent='Open link';a.target='_blank';a.style.color='var(--info)';asinTd.appendChild(a);});}else{asinTd.textContent=row.asin;}
    tr.appendChild(asinTd);
    const dTd=document.createElement('td');dTd.className='td-wrap';dTd.style.cssText='max-width:280px;font-size:.77rem';dTd.textContent=row.description;tr.appendChild(dTd);
    const tTd=document.createElement('td');tTd.style.cssText='color:var(--info);font-weight:700;font-size:.77rem';tTd.textContent=row.trexid;tr.appendChild(tTd);
    const cTd=document.createElement('td');cTd.className='td-wrap';cTd.style.cssText='max-width:260px;font-size:.73rem';
    let last=0,match,text=row.comments;urlRe.lastIndex=0;
    while((match=urlRe.exec(text))!==null){cTd.appendChild(document.createTextNode(text.substring(last,match.index)));const a=document.createElement('a');a.href=match[1];a.textContent=match[1];a.target='_blank';a.style.color='var(--info)';cTd.appendChild(a);last=match.index+match[1].length;}
    cTd.appendChild(document.createTextNode(text.substring(last)));tr.appendChild(cTd);
    const decTd=document.createElement('td');decTd.className='td-wrap';decTd.style.cssText='max-width:140px;font-size:.77rem';decTd.textContent=row.decision;tr.appendChild(decTd);
    tbody.appendChild(tr);
  });
}
